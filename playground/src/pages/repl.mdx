```tsx sandbox autorun lineNumbers=false tabs=false readonly=false preview
import React from 'react'
import { arbitrum } from '@wagmi/core/chains'
import { createConfig, http, getBlockNumber } from '@wagmi/core'

const config = createConfig({
  chains: [arbitrum],
  transports: {
    [arbitrum.id]: http()
  }
})
console.log(config.state.chainId)

export default function App() {
  const [blockNumber, setBlockNumber] = React.useState<string>('')

  React.useEffect(() => getBlockNumber(config).then(setBlockNumber), [])

  return (
    <pre>
      <strong>Block Number:</strong> {blockNumber}
    </pre>
  )
}
```
